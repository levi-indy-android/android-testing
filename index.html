class LoginScreen
  include Gametel
  activity(:LoginScreen)

  text(:username, :index => 0)
  text(:password, :index => 0)

  def login_with(username, password)
    self.username = username
    self.password = password
  end
end

on(LoginScreen).login_with('your_username', '$ecret')
